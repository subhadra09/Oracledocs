select decode(backup_type,'L','ARCHIVELOG','D','FULL-BACKUP','I','INC-BACKUP') "BCK-TYPE", incremental_level "INC-LEVEL", controlfile_included "CTL-INCLUDED", to_char(min(start_time),'DD-MON-YYYY HH24:MI') "START_TIME", to_char(max(completion_time),'DD-MON-YYYY HH24:MI') "COMPLETION_TIME", trunc(sum(elapsed_seconds)/60,2) "ELAP-MINS" from v$BACKUP_SET where trunc(completion_time) >= trunc(sysdate - 5) and backup_type in ('D', 'I') group by backup_type, incremental_level, controlfile_included, trunc(start_time) order by 4;
