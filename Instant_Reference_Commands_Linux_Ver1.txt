------------------------------------------------------
Look at which process is EATING the CPUs:
------------------------------------------------------
# ps -eo pcpu,pid,user,args | sort -k 1 -r | head -10

-------------------------------------------------------
Look at the START time of a process
-------------------------------------------------------
# ps -eo pid,lstart,cmd

-------------------------------------------------------
Total Process COUNT:
-------------------------------------------------------
# ps -ef | wc -l

-------------------------------------------------------
Process count by USER sorted:
-------------------------------------------------------
# ps -eo user=|sort|uniq -c | sort -n -r

-------------------------------------------------------
Look at the number of user processes(threads) 
-------------------------------------------------------
# ps -u testuser -L | wc -l

------------------------------------------------------
Empty LINUX Cache (ROOT command)
------------------------------------------------------
# free -m && sync && echo 3 > /proc/sys/vm/drop_caches && free -m

--------------------------------------------------------
To find processes in 'D' state:
--------------------------------------------------------
# ps auwx | awk '$8 ~ "D"'
